# 交易市场系统设计

## 一、设计理念

### 1.1 核心定位

**交易市场是平台的"经济心脏"——连接创作者与消费者，让价值自由流动。**

一句话理解："买你想玩的，卖你创造的。"

### 1.2 设计原则

| 原则 | 说明 | 实现方式 |
|------|------|---------|
| **低门槛** | 任何人都能买卖 | 一键上架 |
| **高透明** | 价格公开透明 | 实时行情 |
| **强保护** | 交易安全有保障 |  escrow机制 |
| **快流通** | 资产快速变现 | 即时到账 |
| **多品类** | 支持多种资产类型 | 统一交易协议 |

## 三、交易机制

### 3.1 上架流程

```python
class ListingService:
    def create_listing(self, seller_id, asset_id, price):
        """创建上架"""
        # 1. 验证资产所有权
        asset = self.verify_ownership(seller_id, asset_id)
        if not asset:
            raise PermissionError("无权出售此资产")

        # 2. 创建上架记录
        listing = Listing(
            seller_id=seller_id,
            asset_id=asset_id,
            asset_type=asset.type,
            price=price,
            status="active",
            created_at=datetime.now()
        )

        # 3. 锁定资产
        asset.status = "locked"
        asset.listing_id = listing.id

        # 4. 保存上架记录
        self.db.save(listing)

        return listing

    def verify_ownership(self, user_id, asset_id):
        """验证资产所有权"""
        asset = self.db.get_asset(asset_id)
        if asset.owner_id == user_id:
            return asset
        return None
```

### 3.2 购买流程

```python
class PurchaseService:
    def purchase(self, buyer_id, listing_id):
        """购买资产"""
        # 1. 获取上架信息
        listing = self.db.get_listing(listing_id)
        if listing.status != "active":
            raise ValueError("该商品已下架")

        # 2. 检查买家余额
        buyer = self.db.get_user(buyer_id)
        if buyer.balance < listing.price:
            raise ValueError("余额不足")

        # 3. 创建交易订单
        order = Order(
            buyer_id=buyer_id,
            seller_id=listing.seller_id,
            listing_id=listing_id,
            price=listing.price,
            status="pending"
        )

        # 4. 扣除买家资金（进入 escrow）
        buyer.deduct(listing.price)
        self.escrow.hold(order.id, listing.price)

        # 5. 转移资产
        asset = self.db.get_asset(listing.asset_id)
        asset.owner_id = buyer_id
        asset.status = "active"

        # 6. 释放资金给卖家
        seller = self.db.get_user(listing.seller_id)
        seller_share = listing.price * listing.creator_share
        platform_fee = listing.price * listing.platform_fee

        self.escrow.release(order.id, seller.id, seller_share)
        self.revenue.add(platform_fee)

        # 7. 完成订单
        order.status = "completed"
        listing.status = "sold"

        return order
```

### 3.3 拍卖机制

```python
class AuctionService:
    def create_auction(self, seller_id, asset_id, starting_price, duration):
        """创建拍卖"""
        auction = Auction(
            seller_id=seller_id,
            asset_id=asset_id,
            starting_price=starting_price,
            current_price=starting_price,
            highest_bidder=None,
            end_time=datetime.now() + timedelta(hours=duration),
            status="active"
        )
        return auction

    def place_bid(self, bidder_id, auction_id, bid_amount):
        """出价"""
        auction = self.db.get_auction(auction_id)

        if datetime.now() > auction.end_time:
            raise ValueError("拍卖已结束")

        if bid_amount <= auction.current_price:
            raise ValueError("出价必须高于当前价格")

        # 检查 bidder 余额
        bidder = self.db.get_user(bidder_id)
        if bidder.balance < bid_amount:
            raise ValueError("余额不足")

        # 退还上一个出价者的资金
        if auction.highest_bidder:
            last_bidder = self.db.get_user(auction.highest_bidder)
            last_bidder.add(auction.current_price)

        # 扣除新出价者资金
        bidder.deduct(bid_amount)

        # 更新拍卖状态
        auction.current_price = bid_amount
        auction.highest_bidder = bidder_id

        # 延长拍卖时间（如果有新出价）
        if auction.end_time - datetime.now() < timedelta(minutes=5):
            auction.end_time = datetime.now() + timedelta(minutes=5)

    def finalize_auction(self, auction_id):
        """结束拍卖"""
        auction = self.db.get_auction(auction_id)

        if auction.highest_bidder:
            # 有赢家
            winner = self.db.get_user(auction.highest_bidder)
            asset = self.db.get_asset(auction.asset_id)
            asset.owner_id = winner.id

            # 转移资金给卖家
            seller = self.db.get_user(auction.seller_id)
            seller.add(auction.current_price * 0.7)
            self.revenue.add(auction.current_price * 0.3)
        else:
            # 流拍
            asset = self.db.get_asset(auction.asset_id)
            asset.status = "active"

        auction.status = "completed"
```

### 5.2 价格发现

```python
class PriceDiscovery:
    def get_suggested_price(self, asset):
        """获取建议价格"""
        # 1. 基于稀有度
        base_price = self.rarity_base_prices[asset.rarity]

        # 2. 基于属性
        stat_multiplier = self.calculate_stat_multiplier(asset)

        # 3. 基于市场供需
        market_factor = self.get_market_factor(asset.type)

        # 4. 基于历史交易
        historical_avg = self.get_historical_average(asset.type, asset.rarity)

        suggested = (base_price * stat_multiplier * market_factor + historical_avg) / 2

        return suggested

    def get_market_factor(self, asset_type):
        """获取市场供需因子"""
        # 获取该类资产的供需数据
        supply = self.db.count_listings(asset_type)
        demand = self.db.count_purchases(asset_type, days=30)

        if supply == 0:
            return 1.5  # 供不应求

        ratio = demand / supply

        if ratio > 2:
            return 1.5  # 供不应求
        elif ratio > 1:
            return 1.2  # 供需平衡
        elif ratio > 0.5:
            return 0.8  # 供过于求
        else:
            return 0.5  # 严重供过于求
```

## 七、安全机制

### 7.1 交易安全

```python
class TransactionSecurity:
    def verify_transaction(self, order):
        """验证交易"""
        # 1. 验证买家身份
        if not self.verify_identity(order.buyer_id):
            return False

        # 2. 验证卖家身份
        if not self.verify_identity(order.seller_id):
            return False

        # 3. 验证资产合法性
        if not self.verify_asset_legitimacy(order.asset_id):
            return False

        # 4. 检查欺诈风险
        if self.detect_fraud(order):
            return False

        return True

    def detect_fraud(self, order):
        """检测欺诈"""
        # 检查异常交易模式
        buyer = self.db.get_user(order.buyer_id)
        seller = self.db.get_user(order.seller_id)

        # 检查是否自买自卖
        if buyer.id == seller.id:
            return True

        # 检查价格异常
        market_price = self.price_discovery.get_suggested_price(order.asset)
        if order.price > market_price * 10:
            return True

        # 检查交易频率
        recent_transactions = self.db.get_recent_transactions(buyer.id, hours=1)
        if len(recent_transactions) > 10:
            return True

        return False
```

### 7.2 争议处理

```python
class DisputeResolution:
    def create_dispute(self, order_id, reason):
        """创建争议"""
        dispute = Dispute(
            order_id=order_id,
            reason=reason,
            status="open",
            created_at=datetime.now()
        )
        return dispute

    def resolve_dispute(self, dispute_id, resolution):
        """解决争议"""
        dispute = self.db.get_dispute(dispute_id)

        if resolution == "refund_buyer":
            # 退款给买家
            order = self.db.get_order(dispute.order_id)
            buyer = self.db.get_user(order.buyer_id)
            buyer.add(order.price)

            # 收回资产
            asset = self.db.get_asset(order.asset_id)
            asset.owner_id = order.seller_id

        elif resolution == "release_to_seller":
            # 资金释放给卖家
            order = self.db.get_order(dispute.order_id)
            seller = self.db.get_user(order.seller_id)
            seller.add(order.price * 0.7)

        dispute.status = "resolved"
```

## 九、总结

### 9.1 核心设计

```
交易市场 = 低门槛 + 高透明 + 强保护 + 快流通 + 多品类
```

### 9.2 用户旅程

```
创作者：
创建内容 → 上架商品 → 获得收益 → 提现/再投资

消费者：
浏览市场 → 选择商品 → 完成购买 → 使用内容
```

### 9.3 关键指标

| 指标 | 目标 |
|------|------|
| 上架时间 | < 1分钟 |
| 交易完成时间 | < 5秒 |
| 争议率 | < 1% |
| 创作者收益占比 | 70-85% |
