# 社交互联系统设计

## 一、设计理念

### 1.1 核心定位

**社交互联是平台的"网络效应"——让玩家连接彼此，形成社区，增加平台粘性。**

一句话理解："不只是玩游戏，更是和朋友一起创造世界。"

### 1.2 设计原则

| 原则 | 说明 | 实现方式 |
|------|------|---------|
| **低摩擦** | 社交应该自然发生 | 自动推荐+一键添加 |
| **有价值** | 社交带来实际收益 | 组队加成+交易便利 |
| **可展示** | 社交成果可以炫耀 | 个人主页+动态 feed |
| **有归属** | 形成小团体认同 | 公会+圈子+标签 |
| **跨世界** | 社交不受世界限制 | 统一好友系统 |

### 3.2 公会功能

```python
class GuildSystem:
    def create_guild(self, creator_id, guild_name, guild_tag):
        """创建公会"""
        # 检查条件
        creator = self.db.get_user(creator_id)
        if creator.level < 20:
            return False, "需要等级20+"

        if creator.currency < 10000:
            return False, "需要10000金币"

        # 创建公会
        guild = Guild(
            name=guild_name,
            tag=guild_tag,
            leader_id=creator_id,
            level=1,
            exp=0
        )

        # 扣除创建费用
        creator.deduct(10000)

        # 添加创建者为会长
        guild.add_member(creator_id, rank="leader")

        return True, guild

    def join_guild(self, user_id, guild_id):
        """加入公会"""
        guild = self.db.get_guild(guild_id)
        user = self.db.get_user(user_id)

        # 检查是否已满
        if guild.current_members >= guild.max_members:
            return False, "公会已满"

        # 发送申请
        guild.add_application(user_id)

        return True, "申请已发送"

    def guild_war(self, guild_a_id, guild_b_id):
        """公会战"""
        guild_a = self.db.get_guild(guild_a_id)
        guild_b = self.db.get_guild(guild_b_id)

        # 创建公会战
        war = GuildWar(
            guild_a=guild_a_id,
            guild_b=guild_b_id,
            start_time=datetime.now(),
            duration=3600  # 1小时
        )

        # 通知双方成员
        self.notify_guild_members(guild_a_id, "公会战开始！")
        self.notify_guild_members(guild_b_id, "公会战开始！")

        return war
```

## 六、跨世界社交

### 6.1 跨世界机制

```python
class CrossWorldSocial:
    def __init__(self):
        self.global_chat = GlobalChat()
        self.global_friends = GlobalFriends()
        self.global_guilds = GlobalGuilds()

    def send_global_message(self, sender_id, message):
        """发送跨世界消息"""
        # 验证发送者
        sender = self.db.get_user(sender_id)

        # 发送到所有世界
        self.global_chat.broadcast(sender, message)

    def invite_to_world(self, inviter_id, invitee_id, world_id):
        """邀请玩家进入世界"""
        inviter = self.db.get_user(inviter_id)
        invitee = self.db.get_user(invitee_id)
        world = self.db.get_world(world_id)

        # 创建邀请
        invitation = Invitation(
            inviter=inviter_id,
            invitee=invitee_id,
            world=world_id,
            expires_at=datetime.now() + timedelta(minutes=30)
        )

        # 通知被邀请者
        self.notify(invitee_id, f"{inviter.name} 邀请你进入「{world.name}」")

        return invitation

    def join_world(self, user_id, world_id):
        """加入世界"""
        user = self.db.get_user(user_id)
        world = self.db.get_world(world_id)

        # 检查世界访问权限
        if world.visibility == "private":
            if not world.is_invited(user_id):
                return False, "无权访问该世界"

        # 转换角色属性
        converted_character = self.convert_character(user.current_character, world)

        # 进入世界
        user.enter_world(world_id, converted_character)

        # 通知世界内玩家
        self.notify_world_players(world_id, f"{user.name} 进入了世界")

        return True, "进入成功"
```

### 6.2 跨世界交易

```python
class CrossWorldTrading:
    def trade_across_worlds(self, seller_id, buyer_id, item_id):
        """跨世界交易"""
        seller = self.db.get_user(seller_id)
        buyer = self.db.get_user(buyer_id)
        item = self.db.get_item(item_id)

        # 检查物品是否可跨世界交易
        if not item.cross_world_tradable:
            return False, "该物品不可跨世界交易"

        # 创建交易订单
        order = CrossWorldOrder(
            seller=seller_id,
            buyer=buyer_id,
            item=item_id,
            status="pending"
        )

        # 扣除买家资金
        buyer.deduct(item.price)

        # 转移物品
        item.owner_id = buyer_id
        item.world_id = buyer.current_world

        # 给卖家资金
        seller.add(item.price * 0.7)

        order.status = "completed"

        return True, "交易成功"
```

class FriendRecommendation:
    def recommend_friends(self, user_id):
        """推荐好友"""
        user = self.db.get_user(user_id)
        recommendations = []

        # 1. 基于共同世界
        common_world_players = self.find_players_in_same_worlds(user)
        recommendations.extend(common_world_players)

        # 2. 基于共同兴趣
        similar_interest_players = self.find_players_with_similar_interests(user)
        recommendations.extend(similar_interest_players)

        # 3. 基于共同好友
        mutual_friend_players = self.find_players_with_mutual_friends(user)
        recommendations.extend(mutual_friend_players)

        # 4. 基于活跃度
        active_players = self.find_active_players(user)
        recommendations.extend(active_players)

        # 去重并排序
        recommendations = self.deduplicate(recommendations)
        recommendations = self.sort_by_relevance(recommendations)

        return recommendations[:10]

    def find_players_in_same_worlds(self, user):
        """找到在同一世界玩过的玩家"""
        worlds = user.worlds_visited
        players = []

        for world in worlds:
            world_players = self.db.get_world_players(world.id)
            players.extend(world_players)

        return players
```

```
社交互联 = 低摩擦 + 有价值 + 可展示 + 有归属 + 跨世界

    ↓

    ↓
